# 导出sql

```sql
SELECT COALESCE
  ( b.session_id, C.NAME ) AS "用户名",
  A.NAME AS "对话标题",
(SELECT count(*) from messages t where t.conversation_id =a.id)  "消息数",
  A.updated_at AS "更新时间",
    A.created_at AS "创建时间"
FROM
  conversations
  A LEFT JOIN end_users b ON A.from_end_user_id = b."id"
  LEFT JOIN accounts C ON A.from_account_id = C."id" 
WHERE
  A.app_id = '684110a6-9fc5-464d-a985-5c7484c706ea' 
ORDER BY
  A.created_at DESC;
SELECT
  * 
FROM
  workflow_app_logs A 
WHERE
  A.app_id = '684110a6-9fc5-464d-a985-5c7484c706ea';
```
